class Person {
  constructor(n,a,g,h,w){
    this.name=n
    this.age=a
    this.gender=g
    this.height=h
    this.weight=w
  }
  speak(){
    console.log(`hello i am ${this.name} `);
  }
  whatisyourage(){
    console.log(`i am ${this.age} years old `);
  } 
  whatisyourheight(){
    console.log(`i am ${this.height} foot tall`);
  } 
}
